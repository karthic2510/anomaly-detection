Understanding the data¶

The data set available to us consists of 80,000 rows of loan default data. The variable can be bucketed into types such as:

  • Applicant data: Age, education, os_type and so on
  • Credit data: credit checks, credict card limit, debit card score and so on.
  • Call data: calls made, service calls, high risk calls and so on.
  • Loan data: loan term, loan amount, and so on.

We go through the following steps to clean the data:

  1. Rename columns into more understandble and typical terms
  2. Removing unnecessary, repititve columns
  3. Fixing data types by converting categorical data from int64 to categoricals
  4. Creating new features for missing data, missing phone activity and mismatch between application location and IP location
  5. Changing system-based error values like -1, NAs, 99, 999 to 0
  6. Binning continuous variables like loan term, loan amount, age, years of active phone usage.
  7. Cleaning up category variables to include top levels and binning others in other.

We added two additional feature engineering steps this week:

  1. Split application date to day of the week, week of the year, and month of the year.
  2. Use K-means clustering add a feature to the dataset

def data_clean(data1):
    #Fixing the following datatypes to factors
    data1['loan_default'] = data1['loan_default'].astype('object')
    data1['date_applied'] = pd.to_datetime(data1['date_applied'])
    data1['day_applied'] = data1['date_applied'].dt.dayofweek
    data1['week_applied'] = data1['date_applied'].dt.weekofyear
    data1['month_applied'] = data1['date_applied'].dt.month
    data1['gender'] = data1['gender'].astype(str)
    data1['loan_term'] = data1['loan_term'].astype(str)
    data1['education'] = data1['education'].astype(str)
    data1['OS_type'] = data1['OS_type'].astype(str)
    data1['app_city_level'] = data1['app_city_level'].astype(str)
    data1['ip_appl_mismatch'] = data1['ip_appl_mismatch'].astype(str)
    data1['ap009'] = data1['ap009'].astype(str)
    data1['mob_brand'] = data1['mob_brand'].astype(str)
    #creating a new variable mismatch to see if there is a mismatch between the level of the ip address and the application city
    data1['mismatch'] = np.where(data1['app_city_level'] == data1['ip_appl_mismatch'], 1, 0) 
    #dropping columns which do not add any value based on earlier EDA
    drop_cols = ["td044", "td048", "td051", "td054", "td055", "td061", "td062", 'date_applied']
    data1.drop(drop_cols, axis = 1, inplace = True)
    #replacing categories in mobile brands
    data1['mob_brand'] = data1['mob_brand'].str.replace("IPHONE\\d", 'Iphone') #cleaning up iphone levels
    data1['mob_brand'] = data1['mob_brand'].str.replace("IPAD\\d", 'Iphone') #cleaning up iphone levels
    data1['mob_brand'] = data1['mob_brand'].str.lower() #converting all brands to lower
    #other_brands = pd.Series(["web", "iphone", "oppo", "noinfo", "vivo", "huawei", "samsung", "xiaomi", "honor", "meizu", 
    #            "gionee", "leeco", "letv", "360", "coolpad"]) #creating top 15 brands
    #data1['mob_brand'] = data1['mob_brand'].apply(lambda x: x if any([k == x for k in other_brands]) else 'other') #rebucketing all other brands
    #creating a new variable phone missing for missing active phone yeards data
    data1['phone_missing'] = data1['years_phone_active'].apply(lambda x: 1 if np.isnan(x) else 0)
    #creating new variables for rows with missing data
    data1['nodata_1'] = data1['total_in_call_1m'].apply(lambda x: 1 if np.isnan(x) else 0)
    data1['nodata_2'] = data1['count_hit_r6'].apply(lambda x: 1 if np.isnan(x) else 0)
    check_cols = data1.isin([-98, -99, -1]).sum().sort_values(ascending = False)
    check_cols = check_cols[check_cols > 0]
    check_cols = check_cols.index
    for col in check_cols:
        try:
            check_rows = data[data[col].isin([-99, -98, -1])].index
            data1.iloc[check_rows, col] = np.nan
        except:
            continue
    #Creating index of columns with null values
    null_col = data1.isnull().sum().sort_values(ascending = False)
    null_col = null_col[ null_col > 0]
    #Looping through to replace NAs with mean values
    null_col = null_col.index
    for col in null_col:
        data1.loc[:, col].fillna(0, inplace = True)
    return data1

def createGains(model, test_hex, target):
    predictions = model.predict(test_hex)['p1']
    test_scores = test_hex[target].cbind(predictions).as_data_frame()

    #sort on prediction (descending), add id, and decile for groups containing 1/10 of datapoints
    test_scores = test_scores.sort_values(by='p1',ascending=False)
    test_scores['row_id'] = range(0,0+len(test_scores))
    test_scores['decile'] = ( test_scores['row_id'] / (len(test_scores)/10) ).astype(int)
    #see count by decile
    test_scores.loc[test_scores['decile'] == 10]=9
    test_scores['decile'].value_counts()

    #create gains table
    gains = test_scores.groupby('decile')[target].agg(['count','sum'])
    gains.columns = ['count','actual']
    gains

    #add features to gains table
    gains['non_actual'] = gains['count'] - gains['actual']
    gains['cum_count'] = gains['count'].cumsum()
    gains['cum_actual'] = gains['actual'].cumsum()
    gains['cum_non_actual'] = gains['non_actual'].cumsum()
    gains['percent_cum_actual'] = (gains['cum_actual'] / np.max(gains['cum_actual'])).round(2)
    gains['percent_cum_non_actual'] = (gains['cum_non_actual'] / np.max(gains['cum_non_actual'])).round(2)
    gains['if_random'] = np.max(gains['cum_actual']) /10 
    gains['if_random'] = gains['if_random'].cumsum()
    gains['lift'] = (gains['cum_actual'] / gains['if_random']).round(2)
    gains['K_S'] = np.abs( gains['percent_cum_actual'] -  gains['percent_cum_non_actual'] ) * 100
    gains['gain']=(gains['cum_actual']/gains['cum_count']*100).round(2)
    gains = pd.DataFrame(gains)
    return(gains)
